56 void MotionStabilizationPipeline::stabilize(
57  int size, const vector<Mat> &motions, pair<int,int> range, Mat *stabilizationMotions)
58 {
59  vector<Mat> updatedMotions(motions.size());
60  for (size_t i = 0; i < motions.size(); ++i)
61  updatedMotions[i] = motions[i].clone();
62 
63  vector<Mat> stabilizationMotions_(size);
64 
65  for (int i = 0; i < size; ++i)
66  stabilizationMotions[i] = Mat::eye(3, 3, CV_32F);
67 
68  for (size_t i = 0; i < stabilizers_.size(); ++i)
69  {
70  stabilizers_[i]->stabilize(size, updatedMotions, range, &stabilizationMotions_[0]);
71 
72  for (int k = 0; k < size; ++k)
73  stabilizationMotions[k] = stabilizationMotions_[k] * stabilizationMotions[k];
74 
75  for (int j = 0; j + 1 < size; ++j)
76  {
77  Mat S0 = stabilizationMotions[j];
78  Mat S1 = stabilizationMotions[j+1];
79  at(j, updatedMotions) = S1 * at(j, updatedMotions) * S0.inv();

93 void GaussianMotionFilter::setParams(int _radius, float _stdev)
94 {
95  radius_ = _radius;
96  stdev_ = _stdev > 0.f ? _stdev : sqrt(static_cast<float>(_radius));
97 
98  float sum = 0;
99  weight_.resize(2*radius_ + 1);
100  for (int i = -radius_; i <= radius_; ++i)
101  sum += weight_[radius_ + i] = std::exp(-i*i/(stdev_*stdev_));
102  for (int i = -radius_; i <= radius_; ++i)
103  weight_[radius_ + i] /= sum;
104 }

107 Mat GaussianMotionFilter::stabilize(int idx, const vector<Mat> &motions, pair<int,int> range)
108 {
109  const Mat &cur = at(idx, motions);
110  Mat res = Mat::zeros(cur.size(), cur.type());
111  float sum = 0.f;
112  int iMin = max(idx - radius_, range.first);
113  int iMax = min(idx + radius_, range.second);
114  for (int i = iMin; i <= iMax; ++i)
115  {
116  res += weight_[radius_ + i - idx] * getMotion(idx, i, motions);
117  sum += weight_[radius_ + i - idx];
118  }
119  return sum > 0.f ? res / sum : Mat::eye(cur.size(), cur.type());
120 }

589 // Rects must be passed in the [(0,0), (w,0), (w,h), (0,h)] order.
590 static inline bool isRectInside(const Point2f a[4], const Point2f b[4])
591 {
592  for (int i = 0; i < 4; ++i)
593  if (b[i].x > a[0].x && b[i].x < a[2].x && b[i].y > a[0].y && b[i].y < a[2].y)
594  return false;
595  for (int i = 0; i < 4; ++i)
596  for (int j = 0; j < 4; ++j)

598  return false;

604 {
605  Point2f pt[4] = {Point2f(0,0), Point2f(w,0), Point2f(w,h), Point2f(0,h)};
606  Point2f Mpt[4];
607  float z;
608 
609  for (int i = 0; i < 4; ++i)
610  {
611  Mpt[i].x = M[0]*pt[i].x + M[1]*pt[i].y + M[2];
612  Mpt[i].y = M[3]*pt[i].x + M[4]*pt[i].y + M[5];
613  z = M[6]*pt[i].x + M[7]*pt[i].y + M[8];
614  Mpt[i].x /= z;
615  Mpt[i].y /= z;

619  pt[1] = Point2f(w - dx, dy);
620  pt[2] = Point2f(w - dx, h - dy);
621  pt[3] = Point2f(dx, h - dy);

641 Mat ensureInclusionConstraint(const Mat &M, Size size, float trimRatio)
642 {
643  CV_Assert(M.size() == Size(3,3) && M.type() == CV_32F);
644 
645  const float w = static_cast<float>(size.width);
646  const float h = static_cast<float>(size.height);
647  const float dx = floor(w * trimRatio);
648  const float dy = floor(h * trimRatio);
649  const float srcM[] =
650  {M.at<float>(0,0), M.at<float>(0,1), M.at<float>(0,2),
651  M.at<float>(1,0), M.at<float>(1,1), M.at<float>(1,2),
652  M.at<float>(2,0), M.at<float>(2,1), M.at<float>(2,2)};
653 
654  float curM[9];
655  float t = 0;
656  relaxMotion(srcM, t, curM);
657  if (isGoodMotion(curM, w, h, dx, dy))
658  return M;
659 
660  float l = 0, r = 1;
661  while (r - l > 1e-3f)
662  {
663  t = (l + r) * 0.5f;
664  relaxMotion(srcM, t, curM);
665  if (isGoodMotion(curM, w, h, dx, dy))
666  r = t;
667  else
668  l = t;
669  }
670 
671  return (1 - r) * M + r * Mat::eye(3, 3, CV_32F);
