91 Mat StabilizerBase::nextStabilizedFrame()
92 {
93  // check if we've processed all frames already
94  if (curStabilizedPos_ == curPos_ && curStabilizedPos_ != -1)
95  {
96  logProcessingTime();
97  return Mat();
98  }
99 
100  bool processed;
101  do processed = doOneIteration();
102  while (processed && curStabilizedPos_ == -1);
103 
104  // check if the frame source is empty
105  if (curStabilizedPos_ == -1)
106  {
107  logProcessingTime();
108  return Mat();
109  }
110 
111  return postProcessFrame(at(curStabilizedPos_, stabilizedFrames_));
112 }

115 bool StabilizerBase::doOneIteration()
116 {
117  Mat frame = frameSource_->nextFrame();
118  if (!frame.empty())
119  {
120  curPos_++;
121 
122  if (curPos_ > 0)
123  {
124  at(curPos_, frames_) = frame;
125 
126  if (doDeblurring_)
127  at(curPos_, blurrinessRates_) = calcBlurriness(frame);
128 
129  at(curPos_ - 1, motions_) = estimateMotion();
130 
131  if (curPos_ >= radius_)
132  {
133  curStabilizedPos_ = curPos_ - radius_;
134  stabilizeFrame();
135  }
136  }
137  else
138  setUp(frame);
139 
140  log_->print(".");
141  return true;
142  }
143 
144  if (curStabilizedPos_ < curPos_)
145  {
146  curStabilizedPos_++;
147  at(curStabilizedPos_ + radius_, frames_) = at(curPos_, frames_);
148  at(curStabilizedPos_ + radius_ - 1, motions_) = Mat::eye(3, 3, CV_32F);
149  stabilizeFrame();
150 
151  log_->print(".");
152  return true;
153  }
154 
155  return false;
156 }

190 void StabilizerBase::stabilizeFrame()
191 {
192  Mat stabilizationMotion = estimateStabilizationMotion();
193  if (doCorrectionForInclusion_)
194  stabilizationMotion = ensureInclusionConstraint(stabilizationMotion, frameSize_, trimRatio_);
195 
196  at(curStabilizedPos_, stabilizationMotions_) = stabilizationMotion;
197 
198  if (doDeblurring_)
199  {
200  at(curStabilizedPos_, frames_).copyTo(preProcessedFrame_);
201  deblurer_->deblur(curStabilizedPos_, preProcessedFrame_);
202  }
203  else
204  preProcessedFrame_ = at(curStabilizedPos_, frames_);
205 
206  // apply stabilization transformation
207 
208  if (motionEstimator_->motionModel() != MM_HOMOGRAPHY)
209  warpAffine(
210  preProcessedFrame_, at(curStabilizedPos_, stabilizedFrames_),
211  stabilizationMotion(Rect(0,0,3,2)), frameSize_, INTER_LINEAR, borderMode_);
212  else
213  warpPerspective(
214  preProcessedFrame_, at(curStabilizedPos_, stabilizedFrames_),
215  stabilizationMotion, frameSize_, INTER_LINEAR, borderMode_);
216 
217  if (doInpainting_)
218  {
219  if (motionEstimator_->motionModel() != MM_HOMOGRAPHY)
220  warpAffine(
221  frameMask_, at(curStabilizedPos_, stabilizedMasks_),
222  stabilizationMotion(Rect(0,0,3,2)), frameSize_, INTER_NEAREST);
223  else
224  warpPerspective(
225  frameMask_, at(curStabilizedPos_, stabilizedMasks_),
226  stabilizationMotion, frameSize_, INTER_NEAREST);
227 
228  erode(at(curStabilizedPos_, stabilizedMasks_), at(curStabilizedPos_, stabilizedMasks_),
229  Mat());
230 
231  at(curStabilizedPos_, stabilizedMasks_).copyTo(inpaintingMask_);
232 
233  inpainter_->inpaint(
234  curStabilizedPos_, at(curStabilizedPos_, stabilizedFrames_), inpaintingMask_);
235  }
236 }

370 void TwoPassStabilizer::runPrePassIfNecessary()
371 {
372  if (!isPrePassDone_)
373  {
374  // check if we must do wobble suppression
375 
376  WobbleSuppressorBase *wobble = static_cast<WobbleSuppressorBase*>(wobbleSuppressor_);
377  doWobbleSuppression_ = dynamic_cast<NullWobbleSuppressor*>(wobble) == 0;
378 
379  // estimate motions
380 
381  clock_t startTime = clock();
382  log_->print("first pass: estimating motions");
383 
384  Mat prevFrame, frame;
385  bool ok = true, ok2 = true;
386 
387  while (!(frame = frameSource_->nextFrame()).empty())
388  {
389  if (frameCount_ > 0)
390  {
391  motions_.push_back(motionEstimator_->estimate(prevFrame, frame, &ok));
392 
393  if (doWobbleSuppression_)
394  {
395  Mat M = wobbleSuppressor_->motionEstimator()->estimate(prevFrame, frame, &ok2);
396  if (ok2)
397  motions2_.push_back(M);
398  else
399  motions2_.push_back(motions_.back());
400  }
401 
402  if (ok)
403  {
404  if (ok2) log_->print(".");
405  else log_->print("?");
406  }
407  else log_->print("x");
408  }
409  else
410  {
411  frameSize_ = frame.size();
412  frameMask_.create(frameSize_, CV_8U);
413  frameMask_.setTo(255);
414  }
415 
416  prevFrame = frame;
417  frameCount_++;
418  }
419 
420  clock_t elapsedTime = clock() - startTime;
421  log_->print("\nmotion estimation time: %.3f sec\n",
422  static_cast<double>(elapsedTime) / CLOCKS_PER_SEC);
423 
424  // add aux. motions
425 
426  for (int i = 0; i < radius_; ++i)
427  motions_.push_back(Mat::eye(3, 3, CV_32F));
428 
429  // stabilize
430 
431  startTime = clock();
432 
433  stabilizationMotions_.resize(frameCount_);
434  motionStabilizer_->stabilize(
435  frameCount_, motions_, make_pair(0, frameCount_ - 1), &stabilizationMotions_[0]);
436 
437  elapsedTime = clock() - startTime;
438  log_->print("motion stabilization time: %.3f sec\n",
439  static_cast<double>(elapsedTime) / CLOCKS_PER_SEC);
440 
441  // estimate optimal trim ratio if necessary
442 
443  if (mustEstTrimRatio_)
444  {
445  trimRatio_ = 0;
446  for (int i = 0; i < frameCount_; ++i)
447  {
448  Mat S = stabilizationMotions_[i];
449  trimRatio_ = std::max(trimRatio_, estimateOptimalTrimRatio(S, frameSize_));
450  }
451  log_->print("estimated trim ratio: %f\n", static_cast<double>(trimRatio_));
452  }
453 
454 #if SAVE_MOTIONS
455  saveMotions(frameCount_, motions_, stabilizationMotions_);
456 #endif
457 
458  isPrePassDone_ = true;
459  frameSource_->reset();
460  }
461 }
