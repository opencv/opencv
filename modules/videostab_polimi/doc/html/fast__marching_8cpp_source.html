54 float FastMarchingMethod::solve(int x1, int y1, int x2, int y2) const
55 {
56  float sol = inf_;
57  if (y1 >=0 && y1 < flag_.rows && x1 >= 0 && x1 < flag_.cols && flag_(y1,x1) == KNOWN)
58  {
59  float t1 = dist_(y1,x1);
60  if (y2 >=0 && y2 < flag_.rows && x2 >= 0 && x2 < flag_.cols && flag_(y2,x2) == KNOWN)
61  {
62  float t2 = dist_(y2,x2);
63  float r = sqrt(2 - sqr(t1 - t2));
64  float s = (t1 + t2 - r) / 2;
65 
66  if (s >= t1 && s >= t2)
67  sol = s;
68  else
69  {
70  s += r;
71  if (s >= t1 && s >= t2)
72  sol = s;
73  }
74  }
75  else
76  sol = 1 + t1;
77  }
78  else if (y2 >=0 && y2 < flag_.rows && x2 >= 0 && x2 < flag_.cols && flag_(y2,x2) == KNOWN)
79  sol = 1 + dist_(y2,x1);
80  return sol;
81 }

97 void FastMarchingMethod::heapDown(int idx)
98 {
99  int l, r, smallest;
100  for(;;)
101  {
102  l = 2*idx+1;
103  r = 2*idx+2;
104  smallest = idx;
105 
106  if (l < size_ && narrowBand_[l] < narrowBand_[smallest]) smallest = l;
107  if (r < size_ && narrowBand_[r] < narrowBand_[smallest]) smallest = r;
108 
109  if (smallest == idx)
110  break;
111  else
112  {
113  std::swap(indexOf(narrowBand_[idx]), indexOf(narrowBand_[smallest]));
114  std::swap(narrowBand_[idx], narrowBand_[smallest]);
